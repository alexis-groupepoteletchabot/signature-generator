  <script>

    const blobLogo = "blob:https://portal.crossware365.com/a0e88162-a6c7-4661-a431-6e8a58c39d32";


    const formFields = ['firstName', 'lastName', 'jobTitle', 'email', 'phone1', 'phone2', 'brand'];

    formFields.forEach(id => document.getElementById(id).addEventListener('input', updatePreview));


    function updatePreview() {

      const firstName = document.getElementById('firstName').value.trim();

      const lastName = document.getElementById('lastName').value.trim();

      const jobTitle = document.getElementById('jobTitle').value.trim();

      const email = document.getElementById('email').value.trim();

      const phone1 = document.getElementById('phone1').value.trim();

      const phone2 = document.getElementById('phone2').value.trim();


      const signatureHTML = `

        <table cellpadding="0" cellspacing="0" class="signature-table" style="font-family: Georgia, serif; font-size: 13px; line-height: 1.4; color: #000000;">

          <tr>

            <td style="vertical-align: middle; padding-right: 15px;">

              <img src="${blobLogo}" alt="Logo" style="display: block; width: 220px; height: auto; border: none;">

            </td>

            <td style="vertical-align: middle;">

              <div style="font-weight: bold; font-size: 14px; text-transform: uppercase;">${firstName} ${lastName}</div>

              <div style="font-style: italic; font-size: 13px;">${jobTitle}</div>

              <div style="margin-top: 6px; font-size: 13px;">

                <a href="mailto:${email}" style="color: #0033cc; text-decoration: none;">${email}</a>

              </div>

              <div style="margin-top: 4px; font-size: 13px;">

                ${phone1 ? phone1 : ''}${phone1 && phone2 ? ' — ' : ''}${phone2 ? phone2 : ''}

              </div>

            </td>

          </tr>

        </table>

      `;

      document.getElementById('preview').innerHTML = signatureHTML;

    }


    function downloadSignature() {

      const html = document.getElementById('preview').innerHTML;

      const blob = new Blob([html], { type: 'text/html' });

      const link = document.createElement('a');

      link.href = URL.createObjectURL(blob);

      link.download = 'signature.html';

      link.click();

    }


    updatePreview();
